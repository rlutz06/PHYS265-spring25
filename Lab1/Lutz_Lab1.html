def grav_force(M1, m2, x1, y1, x2, y2):
    r_x=x2-x1
    r_y=y2-y1 
    r_squared=(r_x**2)+(r_y**2)
    r=np.sqrt(r_squared)
    r_squared=np.where(r_squared==0,np.nan,r_squared)
    Fmag=-G*M1*m2/r_squared

    Fx=Fmag*r_x/r 
    Fy=Fmag*r_y/r 

    return Fx,Fy

x=np.linspace(-1.5*dmoon, 1.5*dmoon,500) 
y = np.linspace(-1.5*dmoon, 1.5*dmoon,500) 
x_grid, y_grid = np.meshgrid(x,y)

Fx_earth, Fy_earth = grav_force(Mearth,1,xearth,yearth,x_grid,y_grid)
Fx_moon,Fy_moon = grav_force(Mmoon,1,xmoon,ymoon,x_grid,y_grid)

Fx_total=Fx_earth+Fx_moon 
Fy_total=Fy_earth+Fy_moon 

fig, ax4 = plt.subplots(1,1) 

norm = LogNorm(vmin=np.min(Fx_total[Fx_total>0]), vmax=np.max(Fx_total)) 

strm = ax4.streamplot(x_grid, y_grid, Fx_total, Fy_total, color=np.sqrt(Fx_total**2+Fy_total**2), norm=norm)

ax4.set_title("Gravitation Force Field of Earth-Moon System")
ax4.set_xlabel("X(m)") 
ax4.set_ylabel("Y(m)") 
ax4.set_aspect('equal') 

cbar = fig.colorbar(strm.lines, ax=ax4) 
cbar.set_label("Gravitational Force Magnitude (N)") 
fig.tight_layout()

from scipy.integrate import quad

m0=2.8*10**6
mf=7.5*10**5
mdot=1.3*10**4

T = (m0-mf)/mdot
print(f'Burn time: {T} seconds') 

def delta_v(t,m0,mdot,ve,g):
    m=m0-mdot*t
    if m<mf:
        m=mf 
    return ve*np.log(m0/m)-g*t


def altitude(t,m0,mf,mdot,ve,g): 
    return delta_v(t,m0,mdot,ve,g) 

altitude, error = quad(altitude, 0, T, args=(m0,mf,mdot,2.4*10**3,9.81))
print(f'Altitude: {altitude} meters')
